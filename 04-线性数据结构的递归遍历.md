## 04-线性数据结构的递归遍历

### 数组的递归遍历
```js
function ArrTraverse1(arr, i) { // 数组 开始遍历位置的索引
  if (arr[i] == null) return;
  console.log(arr[i]);
  ArrTraverse1(arr, i + 1);
}
```


### // 链表的递归遍历
```js
function linkTraverse1 (root) {
  if (root == null ) return;
  console.log(root.value);
  linkTraverse1(root.next);
}
```
